<script>
(function () {
  var TOTAL  = 219;
  var PRELOAD = 2;
  var BASE   = "{{ '/assets/img/triton-tutorial/slide-' | relative_url }}";

  // Chapter boundaries: [startSlide, navItem]
  var CHAPTERS = [
    { start: 1,   el: null },
    { start: 2,   el: null },
    { start: 15,  el: null },
    { start: 55,  el: null },
    { start: 93,  el: null },
    { start: 139, el: null },
    { start: 181, el: null },
  ];

  var current = 1;
  var images  = {};

  var stage       = document.getElementById('slide-stage');
  var counter     = document.getElementById('slide-counter');
  var bar         = document.getElementById('progress-bar');
  var btnPrev     = document.getElementById('btn-prev');
  var btnNext     = document.getElementById('btn-next');
  var placeholder = document.getElementById('slide-placeholder');
  var navItems    = document.querySelectorAll('#slide-nav li');

  // Wire chapter nav elements
  navItems.forEach(function (li, i) {
    CHAPTERS[i].el = li;
    li.addEventListener('click', function () {
      showSlide(CHAPTERS[i].start);
    });
  });

  function pad(n) { return String(n).padStart(3, '0'); }

  function loadSlide(n) {
    if (n < 1 || n > TOTAL || images[n]) return;
    var img = document.createElement('img');
    img.alt = 'Slide ' + n;
    img.src = BASE + pad(n) + '.jpg';
    images[n] = img;
    stage.appendChild(img);
  }

  function activeChapterFor(n) {
    var idx = 0;
    for (var i = 0; i < CHAPTERS.length; i++) {
      if (n >= CHAPTERS[i].start) idx = i;
    }
    return idx;
  }

  function updateNav(n) {
    var active = activeChapterFor(n);
    navItems.forEach(function (li, i) {
      li.classList.toggle('active', i === active);
    });
  }

  function showSlide(n) {
    if (images[current]) images[current].classList.remove('active');

    current = n;
    loadSlide(current);
    for (var i = 1; i <= PRELOAD; i++) {
      loadSlide(current + i);
      loadSlide(current - i);
    }

    var img = images[current];
    if (img.complete && img.naturalWidth > 0) {
      placeholder.style.display = 'none';
      img.classList.add('active');
    } else {
      placeholder.style.display = 'flex';
      img.onload = function () {
        placeholder.style.display = 'none';
        img.classList.add('active');
      };
    }

    counter.textContent = current + ' / ' + TOTAL;
    bar.style.width = (current / TOTAL * 100).toFixed(2) + '%';
    btnPrev.disabled = current === 1;
    btnNext.disabled = current === TOTAL;
    updateNav(current);
  }

  btnPrev.addEventListener('click', function () { if (current > 1) showSlide(current - 1); });
  btnNext.addEventListener('click', function () { if (current < TOTAL) showSlide(current + 1); });
  stage.addEventListener('click',   function () { if (current < TOTAL) showSlide(current + 1); });

  document.addEventListener('keydown', function (e) {
    if (e.key === 'ArrowRight' || e.key === 'ArrowDown') {
      if (current < TOTAL) showSlide(current + 1);
    } else if (e.key === 'ArrowLeft' || e.key === 'ArrowUp') {
      if (current > 1) showSlide(current - 1);
    }
  });

  showSlide(1);
})();
</script>
